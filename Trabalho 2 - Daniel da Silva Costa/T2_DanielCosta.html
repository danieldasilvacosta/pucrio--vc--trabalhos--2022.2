def calcula_erro( a, b, c, d ):

    soma = 0.0
    n = x.shape[0]
    for i in range(n):

        e = y[i] - ( a + b*x[i] + c*(x[i]**2) + d*(x[i]**3) )
        soma += e * e

    # return soma/n # Sem dividir pelo N (quantidade de elementos) a funo  ajustada mais rapidamente.
    return soma

# Baseado em https://pytorch.org/tutorials/beginner/pytorch_with_examples.html

def grad_desc( a0, b0, c0, d0, tol, max_it, learning_rate ):

    erro0 = calcula_erro( a0, b0, c0, d0 )
    erros = [erro0]

    for t in range(max_it):
      
        # Forward pass: compute predicted y
        y_pred = a0 + b0 * x + c0 * x ** 2 + d0 * x ** 3

        # Compute and print loss
        loss = calcula_erro( a0, b0, c0, d0 )

        # Backprop to compute gradients of a, b, c, d with respect to loss
        grad_y_pred = 2.0 * (y_pred - y)
        grad_a = grad_y_pred.sum()
        grad_b = (grad_y_pred * x).sum()
        grad_c = (grad_y_pred * x ** 2).sum()
        grad_d = (grad_y_pred * x ** 3).sum()

        # Update weights
        a1 = a0 - learning_rate * grad_a
        b1 = b0 - learning_rate * grad_b
        c1 = c0 - learning_rate * grad_c
        d1 = d0 - learning_rate * grad_d

        a0 = a1
        b0 = b1
        c0 = c1
        d0 = d1

        erro1 = calcula_erro( a1, b1, c1, d1 )
        erros.append( erro1 )

    print( f'Soluo a = {a1:.3f}' )
    print( f'Soluo b = {b1:.3f}' )
    print( f'Soluo c = {c1:.3f}' )
    print( f'Soluo d = {d1:.3f}' )

    return a1, b1, c1, d1, np.array(erros)

learning_rate = 0.001
iteracoes = 10000

def forward(x):
    return a + b*x + c*(x**2) + d*(x**3)

def loss(y, y_pred):
    return (y - y_pred).pow(2).sum()
    # return (y - y_pred).pow(2).sum() / len(y) # Sem dividir pelo N (quantidade de elementos) a funo  ajustada mais rapidamente.

for iteracao in range(0, iteracoes):

    y_pred = forward(x)

    l = loss(y, y_pred)
    
    # Calcular os gradientes de forma automtica usando o PyTorch (autograd)
    l.backward()

    with torch.no_grad():

        a -= learning_rate * a.grad
        b -= learning_rate * b.grad
        c -= learning_rate * c.grad
        d -= learning_rate * d.grad

    b.grad.zero_()
